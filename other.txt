Check number of articles per category:

    SELECT 
    nc.category,
nc.id_category,
    COUNT(na.id_article) as article_count
FROM 
    news_categories nc
LEFT JOIN 
    news_articles na ON nc.id_category = na.id_category
GROUP BY 
    nc.id_category, nc.category
ORDER BY 
id_category;
